local success, err = pcall(function()
    loadstring(game:HttpGet("https://raw.githubusercontent.com/mopsscript7-gif/loader1/refs/heads/main/loader.txt"))()
end)

if not success then
    game:Shutdown()
    wait()
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = "Error",
        Text = "Failed to load script: " .. tostring(err),
        Duration = 5
    })
end
